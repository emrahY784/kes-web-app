async function calculate() { const country = document.getElementById('country').value; const year = parseInt(document.getElementById('year').value); const alpha = parseFloat(document.getElementById('alpha').value); const beta = parseFloat(document.getElementById('beta').value); const gamma = parseFloat(document.getElementById('gamma').value); const lambd = parseFloat(document.getElementById('lambd').value); const geometric = document.getElementById('geometric').checked; const use_min = document.getElementById('use_min').checked; const resultDiv = document.getElementById('result'); resultDiv.innerHTML = 'Hesaplanıyor...'; try { const res = await fetch('/calculate', { method: 'POST', headers: { 'Content-Type': 'application/json' }, body: JSON.stringify({ country, year, alpha, beta, gamma, lambd, geometric, use_min }) }); const text = await res.text(); // Önce text olarak al let result; try { result = JSON.parse(text); } catch(e) { resultDiv.innerHTML = `

Sunucu yanıtı JSON değil. İlk 200 karakter: ${text.substring(0,200)}

`; return; } if (result.error) { resultDiv.innerHTML = `

${result.error}

`; } else { let yearMsg = ''; if (result.used_year !== result.requested_year) { yearMsg = `

⚠️ İstenen yıl (${result.requested_year}) için veri yoktu. En yakın yıl (${result.used_year}) kullanıldı.

`; } resultDiv.innerHTML = ` ${yearMsg}

KES: ${result.kes} (${result.interpretation})

V_iç Skoru: ${result.v_ic_score}

Yorum: 0-33 Sermayeci, 34-66 Karma, 67-100 Kamucu

`; } } catch (err) { resultDiv.innerHTML = `

Ağ hatası: ${err.message}

`; } }
